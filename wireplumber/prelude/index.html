use wireplumber::prelude::*;
use wireplumber::pw::{self, PipewireObject};

fn get_id(obj: &PipewireObject) -> u32 {
  obj.pw_property(pw::PW_KEY_OBJECT_ID)
    .expect("how do you not know who you are")
}
